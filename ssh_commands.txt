# Выполните эти команды на сервере:

# 1. Проверьте, где вы находитесь
pwd

# 2. Посмотрите, что в папке
ls -la

# 3. Если папка пустая или там нет файлов проекта:
cd ~
rm -rf finance-bot  # удалите пустую папку
git clone https://github.com/YOUR_USERNAME/YOUR_REPO.git finance-bot
cd finance-bot

# 4. Скопируйте env.example в .env
cp env.example .env

# 5. Отредактируйте .env
nano .env

# 6. Сделайте скрипт исполняемым и запустите
chmod +x deploy.sh
./deploy.sh prod
